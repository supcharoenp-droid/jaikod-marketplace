### р╕зр╕┤р╕Шр╕╡р╕Чр╕╡р╣И 2: р╣Гр╕Кр╣Й Script (р╕кр╕│р╕лр╕гр╕▒р╕Ъ Developer)

```javascript
const serviceAccount = require('./serviceAccountKey.json');

admin.initializeApp({
  credential: admin.credential.cert(serviceAccount)
});

const db = admin.firestore();
const auth = admin.auth();

async function createAdmin(email, password, displayName, role) {
  try {
    // 1. Create Auth User
    const userRecord = await auth.createUser({
      email: email,
      password: password,
      displayName: displayName
    });
    
    console.log('тЬЕ Created user:', userRecord.uid);
    
    // 2. Create Admin Document
    await db.collection('admins').doc(userRecord.uid).set({
      email: email,
      displayName: displayName,
      role: role,
      permissions: [],
      is_active: true,
      created_at: admin.firestore.FieldValue.serverTimestamp(),
      created_by: 'system'
    });
    
    console.log('тЬЕ Created admin document');
    console.log('ЁЯОЙ Admin created successfully!');
    console.log('ЁЯУз Email:', email);
    console.log('ЁЯФС Password:', password);
    console.log('ЁЯСд Role:', role);
    
  } catch (error) {
    console.error('тЭМ Error:', error);
  }
}

// р╕кр╕гр╣Йр╕▓р╕З Admin р╣Бр╕Хр╣Ир╕ер╕░р╕гр╕░р╕Фр╕▒р╕Ъ
async function createAllAdmins() {
  await createAdmin('superadmin@jaikod.com', 'SuperAdmin123!', 'Super Admin', 'super_admin');
  await createAdmin('manager@jaikod.com', 'Manager123!', 'Admin Manager', 'admin_manager');
  await createAdmin('operations@jaikod.com', 'Operations123!', 'Operations Admin', 'operations_admin');
  await createAdmin('finance@jaikod.com', 'Finance123!', 'Finance Admin', 'finance_admin');
  await createAdmin('moderator@jaikod.com', 'Moderator123!', 'Content Moderator', 'content_moderator');
  await createAdmin('analyst@jaikod.com', 'Analyst123!', 'Data Analyst', 'data_analyst');
  await createAdmin('support@jaikod.com', 'Support123!', 'Customer Support', 'customer_support');
  
  process.exit(0);
}

createAllAdmins();
```
